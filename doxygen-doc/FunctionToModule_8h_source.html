51 Function *globalInitUsesExternalBA(GlobalVariable *GV) {
52  if (!GV->hasInitializer())
53  return nullptr;
54 
55  Constant *I = GV->getInitializer();
56 
57  // walk the values used by the initializer
58  // (and recurse into things like ConstantExpr)
59  std::vector<Constant *> Todo;
60  std::set<Constant *> Done;
61  Todo.push_back(I);
62 
63  while (!Todo.empty()) {
64  Constant *V = Todo.back();
65  Todo.pop_back();
66  Done.insert(V);
67 
68  if (BlockAddress *BA = dyn_cast<BlockAddress>(V)) {
69  Function *F = BA->getFunction();
70  if (F->isDeclaration())
71  return F;
72  }
73 
74  for (User::op_iterator i = V->op_begin(), e = V->op_end(); i != e; ++i) {
75  Constant *C = dyn_cast<Constant>(*i);
76  if (C && !isa<GlobalValue>(C) && !Done.count(C))
77  Todo.push_back(C);
78  }
79  }
80  return nullptr;
81 }

84 static void eliminateAliases(GlobalValue *GV) {
85  // First, check whether a GlobalAlias references this definition.
86  // GlobalAlias MAY NOT reference declarations.
87  for (;;) {
88  // 1. Find aliases
89  SmallVector<GlobalAlias *, 1> aliases;
90  Module *M = GV->getParent();
91  for (Module::alias_iterator I = M->alias_begin(), E = M->alias_end();
92  I != E; ++I)
93  if (I->getAliasee()->stripPointerCasts() == GV)
94  aliases.push_back(&*I);
95  if (aliases.empty())
96  break;
97  // 2. Resolve aliases
98  for (unsigned i = 0, e = aliases.size(); i < e; ++i) {
99  aliases[i]->replaceAllUsesWith(aliases[i]->getAliasee());
100  aliases[i]->eraseFromParent();
101  }
102  // 3. Repeat until no more aliases found; there might
103  // be an alias to an alias...
104  }
105 }

145 static Constant *GetTorInit(std::vector<std::pair<Function *, int>> &TorList) {
146  assert(!TorList.empty() && "Don't create empty tor list!");
147  std::vector<Constant *> ArrayElts;
148  Type *Int32Ty = Type::getInt32Ty(TorList[0].first->getContext());
149 
150  StructType *STy = StructType::get(Int32Ty, TorList[0].first->getType());
151  for (unsigned i = 0, e = TorList.size(); i != e; ++i) {
152  Constant *Elts[] = {ConstantInt::get(Int32Ty, TorList[i].second),
153  TorList[i].first};
154  ArrayElts.push_back(ConstantStruct::get(STy, Elts));
155  }
156  return ConstantArray::get(
157  ArrayType::get(ArrayElts[0]->getType(), ArrayElts.size()), ArrayElts);
158 }

164 static void SplitStaticCtorDtor(const char *GlobalName, Module *M1, Module *M2,
165  ValueToValueMapTy &VMap) {
166  GlobalVariable *GV = M1->getNamedGlobal(GlobalName);
167  if (!GV || GV->isDeclaration() || GV->hasLocalLinkage() || !GV->use_empty())
168  return;
169 
170  std::vector<std::pair<Function *, int>> M1Tors, M2Tors;
171  ConstantArray *InitList = dyn_cast<ConstantArray>(GV->getInitializer());
172  if (!InitList)
173  return;
174 
175  for (unsigned i = 0, e = InitList->getNumOperands(); i != e; ++i) {
176  if (ConstantStruct *CS =
177  dyn_cast<ConstantStruct>(InitList->getOperand(i))) {
178  if (CS->getNumOperands() != 2)
179  return; // Not array of 2-element structs.
180 
181  if (CS->getOperand(1)->isNullValue())
182  break; // Found a null terminator, stop here.
183 
184  ConstantInt *CI = dyn_cast<ConstantInt>(CS->getOperand(0));
185  int Priority = CI ? CI->getSExtValue() : 0;
186 
187  Constant *FP = CS->getOperand(1);
188  if (ConstantExpr *CE = dyn_cast<ConstantExpr>(FP))
189  if (CE->isCast())
190  FP = CE->getOperand(0);
191  if (Function *F = dyn_cast<Function>(FP)) {
192  if (!F->isDeclaration())
193  M1Tors.push_back(std::make_pair(F, Priority));
194  else {
195  // Map to M2's version of the function.
196  F = cast<Function>(VMap[F]);
197  M2Tors.push_back(std::make_pair(F, Priority));
198  }
199  }
200  }
201  }
202 
203  GV->eraseFromParent();
204  if (!M1Tors.empty()) {
205  Constant *M1Init = GetTorInit(M1Tors);
206  new GlobalVariable(*M1, M1Init->getType(), false,
207  GlobalValue::AppendingLinkage, M1Init, GlobalName);
208  }
209 
210  GV = M2->getNamedGlobal(GlobalName);
211  assert(GV && "Not a clone of M1?");
212  assert(GV->use_empty() && "llvm.ctors shouldn't have uses!");
213 
214  GV->eraseFromParent();
215  if (!M2Tors.empty()) {
216  Constant *M2Init = GetTorInit(M2Tors);
217  new GlobalVariable(*M2, M2Init->getType(), false,
218  GlobalValue::AppendingLinkage, M2Init, GlobalName);
219  }
220 }

303 martSplitFunctionsOutOfModule(Module *M, const std::string &FName) {
304  // Make sure functions & globals are all external so that linkage
305  // between the two modules will work.
306  for (Module::iterator I = M->begin(), E = M->end(); I != E; ++I)
307  I->setLinkage(GlobalValue::ExternalLinkage);
308  for (Module::global_iterator I = M->global_begin(), E = M->global_end();
309  I != E; ++I) {
310  if (I->hasName() && I->getName()[0] == '\01')
311  I->setName(I->getName().substr(1));
312  I->setLinkage(GlobalValue::ExternalLinkage);
313  }
314 
315  ValueToValueMapTy NewVMap;
316 #if (LLVM_VERSION_MAJOR >= 8) // && (LLVM_VERSION_MINOR < 5)
317  std::unique_ptr<Module> New = CloneModule(*M, NewVMap);
318 #elif (LLVM_VERSION_MAJOR <= 3) && (LLVM_VERSION_MINOR < 8)
319  std::unique_ptr<Module> New(CloneModule(M, NewVMap));
320 #else
321  std::unique_ptr<Module> New = CloneModule(M, NewVMap);
322 #endif
323 
324  // Remove the Test functions from the Safe module
325  std::set<Function *> TestFunctions;
326  Function *F = cast<Function>(M->getFunction(FName));
327  for (Function &I : *M) {
328  if (&I != F) {
329  MART_DEBUG(errs() << "Removing function ");
330 #if (LLVM_VERSION_MAJOR <= 3) && (LLVM_VERSION_MINOR < 5)
331  MART_DEBUG(I.print(errs()));
332 #else
333  MART_DEBUG(I.printAsOperand(errs(), false));
334 #endif
335  MART_DEBUG(errs() << "\n");
336  DeleteFunctionBody(&I); // Function is now external in this module!
337  } else {
338  TestFunctions.insert(cast<Function>(NewVMap[&I]));
339  }
340  }
341 
342  // Remove the Safe functions from the Test module
343  for (Function &I : *New)
344  if (TestFunctions.count(&I))
345  DeleteFunctionBody(&I);
346 
347 // Try to split the global initializers evenly
348 #if (LLVM_VERSION_MAJOR <= 3) && (LLVM_VERSION_MINOR < 5)
349  for (auto glIt = M->global_begin(), glE = M->global_end(); glIt != glE;
350  ++glIt) {
351  GlobalVariable &I = *glIt;
352 #else
353  for (GlobalVariable &I : M->globals()) {
354 #endif
355  GlobalVariable *GV = cast<GlobalVariable>(NewVMap[&I]);
356  if (Function *TestFn = globalInitUsesExternalBA(&I)) {
357  if (Function *SafeFn = globalInitUsesExternalBA(GV)) {
358  errs() << "*** Error: when reducing functions, encountered "
359  "the global '";
360 #if (LLVM_VERSION_MAJOR <= 3) && (LLVM_VERSION_MINOR < 5)
361  GV->print(errs());
362 #else
363  GV->printAsOperand(errs(), false);
364 #endif

366  "from safe function '"
367  << SafeFn->getName() << "' and from test function '"
368  << TestFn->getName() << "'.\n";
369  exit(1);
370  }
371 #if (LLVM_VERSION_MAJOR <= 3) && (LLVM_VERSION_MINOR < 5)

374  // If we keep it in the safe module, then delete it in the test module
