817  index = 0;

819  } else if (!isSequentialType(ptrOprdElemType)) {
820  if (llvm::isa<llvm::LoadInst>(ptrOprd)) {
821  // return the first index (idx = 0)
822  index = 0;
823  return *(gep->idx_begin() + index);
824  } else {
825  return nullptr;
826  }
827  } else {
828  // The pointer operand point to a sequential type, take idx 0 if
829  // comes from load, else take idx 1
830  if (llvm::isa<llvm::LoadInst>(ptrOprd)) {
831  // return the first index (idx = 0)
832  index = 0;
833  return *(gep->idx_begin() + index);
834  } else {
835  // return the first index (idx = 1)
836  if (gep->getNumIndices() > 1) {
837  index = 1;
838  return *(gep->idx_begin() + index);
839  } else {
840  assert(gep->getNumIndices() == 1 &&
841  "gep should have at least 1 index");
842  // Only alloca of an array of array can have this gep
843  // a
844  if (llvm::isa<llvm::AllocaInst>(ptrOprd)) {
845  index = 0;
846  return *(gep->idx_begin() + index);
847  } else {
848  assert (false && "Must be Alloca here (array of array)");
849  }
850  }
851  }
852  }
853  // Should never reach here but add this to suppress build warnings
854  return nullptr;
